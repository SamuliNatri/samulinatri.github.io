Win32 - Drawing Pixels (C)

How to draw a pixel using Windows API Jun 28, 2023 
// main.c
// cl main.c /link user32.lib gdi32.lib

#include <windows.h>
#include <stdint.h>

typedef uint32_t u32;
void *mem;

u32 bitmap_width;
u32 bitmap_height;
u32 client_width;
u32 client_height;

// draws a pixel at X, Y (from top left corner)
void draw_pixel(int x, int y, u32 color) {
    u32 *p = (u32 *)mem;
    p += y * bitmap_width + x;
    *p = color;
}

void clear_screen(u32 color) {
    u32 *p = (u32 *)mem;
    for(u32 i = 0; i < bitmap_width * bitmap_height; ++i) { 
        *p++ = color; 
    }
}

LRESULT CALLBACK wproc(HWND window, UINT message, WPARAM w_param, LPARAM l_param) {
    switch(message) {
        case WM_KEYDOWN: {
            switch(w_param) { 
                case 'O': { 
                    DestroyWindow(window); 
                }; 
            }
        } break;
        case WM_DESTROY: { 
            PostQuitMessage(0); 
        } break;
        default: { 
            return DefWindowProc(window, message, w_param, l_param); 
        }
    }
    return 0;
}

int WINAPI wWinMain(HINSTANCE instance, HINSTANCE prev_instance, PWSTR cmd_line, int cmd_show) {
    
    WNDCLASS wc = {0};
    const char class_name[] = "MyWindowClass";
    wc.lpfnWndProc = wproc;
    wc.hInstance = instance;
    wc.lpszClassName = class_name;
    wc.hCursor = LoadCursor(0, IDC_CROSS);
    
    if(!RegisterClass(&wc)) {
        MessageBox(0, "RegisterClass failed", 0, 0);
        return GetLastError();
    }
    
    HWND window = CreateWindowEx(0, class_name, "Program",
                                    WS_OVERLAPPEDWINDOW|WS_VISIBLE,
                                    CW_USEDEFAULT, CW_USEDEFAULT,
                                    CW_USEDEFAULT, CW_USEDEFAULT,
                                    0, 0, instance, 0);
    
    if(!window) {
        MessageBox(0, "CreateWindowEx failed", 0, 0);
        return GetLastError();
    }
    
    
    // get client area dimensions
    RECT client_rect;
    GetClientRect(window, &client_rect);
    client_width = client_rect.right - client_rect.left;
    client_height = client_rect.bottom - client_rect.top;
    
    bitmap_width = client_width;
    bitmap_height = client_height;
    
    // allocate memory for the bitmap
    mem = VirtualAlloc(0,
                        bitmap_width * bitmap_height * 4, // 4: bytes per pixel
                        MEM_RESERVE|MEM_COMMIT,
                        PAGE_READWRITE
                        );
    
    // bitmap_info struct for StretchDIBits
    BITMAPINFO bi;
    bi.bmiHeader.biSize = sizeof(bi.bmiHeader);
    bi.bmiHeader.biWidth = bitmap_width;
    // negative height makes top left as the coordinate system origin for the draw_pixel() function, otherwise its bottom left
    bi.bmiHeader.biHeight = -bitmap_height;
    bi.bmiHeader.biPlanes = 1;
    bi.bmiHeader.biBitCount = 32;
    bi.bmiHeader.biCompression = BI_RGB;
    
    HDC hdc = GetDC(window);
    
    int running = 1;
    
    while(running) {
        MSG msg;
        while(PeekMessage(&msg, NULL, 0, 0, PM_REMOVE)) {
            if(msg.message == WM_QUIT) running = 0;
            TranslateMessage(&msg);
            DispatchMessage(&msg);
        }
        
        clear_screen(0x333333);
        draw_pixel(100, 100, 0xffffff);
        
        StretchDIBits(hdc, 
                        0, 0,
                        bitmap_width, bitmap_height,
                        0, 0,
                        client_width, client_height,
                        mem, &bi,
                        DIB_RGB_COLORS, SRCCOPY);
    }
    
    return 0;
}
